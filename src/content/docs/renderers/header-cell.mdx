---
title: Header Cell
description: This component renders the cells of the table.
---

## Type definitions

```tsx
import type { HeaderGroup } from "@tanstack/react-table"

export type HeaderComponent<TData> = FC<{
  children: ReactNode
  headerGroups: HeaderGroup<TData>[]
}>
```

## Usage

```tsx
import type { ComponentProps } from "react"

import type { HeaderComponent } from "../../customtypes"

const HeaderRenderer = <TData extends any>({
  children,
  renderer: Renderer,
  headerGroups,
}: ComponentProps<HeaderComponent<TData>> & {
  renderer?: HeaderComponent<TData>
}) => {
  if (Renderer)
    return <Renderer headerGroups={headerGroups}>{children}</Renderer>
  return <thead>{children}</thead>
}
```
